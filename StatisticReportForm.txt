Private Sub UserForm_Initialize()
	degreesOfFreedomLabel.Caption = degreesOfFreedom
	sAposterioriLabel.Caption = sAposteriori
	Label6.Caption = index

	If F_test < F_Fisher Then
		str_F_test = Str(F_test)
		str_F_Fisher = Str(F_Fisher)

		fTestResultLabel.Caption = str_F_test & "<" & str_F_Fisher
		hintLabel.Caption = "F-test is successful."
	Else
		fTestResultLabel.Caption = CStr(F_test) & ">" & CDbl(F_Fisher)
		hintLabel.Caption = "F-test is not successful. Please try a tranformation of higher degree."
	End If

	For i = 1 To nn
		indexListBox.AddItem(i)
		xListBox.AddItem(xvect(i))
		yListBox.AddItem(yvect(i))
		dxListBox.AddItem(Dx(i))
		dyListBox.AddItem(Dy(i))
		sseListBox.AddItem(sse(i))
	Next i
End Sub
