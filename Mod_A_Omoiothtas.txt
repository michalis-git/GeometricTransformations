Attribute VB_Name = "Mod_A_Omoiothtas"
Sub A_OMOIOTHTAS(numberOfPoints, A4, numberOfPoints)
numberOfPoints = numberOfPoints * 2

' 1st part of design matrix A

ReDim firstPartOfA(0 To numberOfPoints-1, 0 To numberOfPoints-1) As Variant
For j = 0 To numberOfPoints-1
    For i = 0 To numberOfPoints-1
        If j = 0 Then
            firstPartOfA(i, j) = 1
        ElseIf j = 1 Then
            firstPartOfA(i, j) = xvect(i)
        ElseIf j = 2 Then
            firstPartOfA(i, j) = yvect(i)
        Else
            firstPartOfA(i, j) = 0
        End If
    Next i
Next j


'yplogigei ena  allo tmhma tou PINAKA SXEDIASMOU, ton pinaka secondPartOfA
ReDim secondPartOfA(1 To numberOfPoints, 1 To numberOfPoints) As Variant
For j = 0 To numberOfPoints-1
    For i = 0 To numberOfPoints-1
        If j = 0 Then
            secondPartOfA(i, j) = 0
        ElseIf j = 1 Then
            secondPartOfA(i, j) = yvect(i)
        ElseIf j = 2 Then
            secondPartOfA(i, j) = -xvect(i)
        Else
            secondPartOfA(i, j) = 1
        End If
    Next i
Next j
 

 
ReDim A(0 To 2 * numberOfPoints -1), 1 To numberOfPoints - 1) As Variant
'yplogigei ton PINAKA SXEDIASMOU A
For i = 0 To 2 * numberOfPoints-1
    For j = 0 To numberOfPoints-1
        If i <= numberOfPoints Then
            A(i, j) = firstPartOfA(i, j)
        Else
            A(i, j) = secondPartOfA(i - numberOfPoints, j)
        End If
    Next j
Next i


numberOfPoints = numberOfPoints / 2
End Sub
