Attribute VB_Name = "ModMULTIPLICATION"
Sub Multiplication(System_DIM1, System_DIM2, System_DIM3, Matrix_Mult() As Variant, M1() As Variant, M2() As Variant)


'Calculates the product [A]*[A-1] which must be always equal to the Singular Matrix [I]
'The same result must also come up for the product: [A-1]*[A]=[I]
'You can try it using: Matrix_Mult(k, m) = Matrix_Mult(k, m) + Inverse_Matrix(L, m) * Matrix_A(k, L)

Erase Matrix_Mult

For k = 1 To System_DIM1
    For m = 1 To System_DIM2
        For l = 1 To System_DIM3
            Matrix_Mult(k, m) = Matrix_Mult(k, m) + M1(k, l) * M2(l, m)
        Next l
    Next m
Next k
Open "c:\Matrix_Mult.txt" For Output As #11
For i = 1 To parameters
    For j = 1 To n
    Print #11, Matrix_Mult(i, j),
    Next j
    Print #11,    ' Print blank line to file.
Next i
Close #11

End Sub
For k = 1 To System_DIM
    For m = 1 To System_DIM
        For l = 1 To System_DIM
            Matrix_Mult(k, m) = Matrix_Mult(k, m) + Matrix_A(k, l) * Inverse_Matrix(l, m)
        Next l
    Next m
Next k
